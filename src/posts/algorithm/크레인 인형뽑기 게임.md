```javascript
function solution(board, moves) {
    var answer = 0
    var stack_box = [0]
    for (var i = 0; i < moves.length; i++) {
        for (var j = 0; j < board.length; j++) {
            var doll = board[j][moves[i]]
            if (doll !== 0) {
                if (stack_box[-1] === doll) {
                    stack_box.pop()
                    answer += 2
                } else {
                    stack_box.push(doll)
                }
                board[j][i] = 0
                break
            }
        }
    }
    return answer
}
```
